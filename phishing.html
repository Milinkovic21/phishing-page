const express = require('express'); const bodyParser = require('body-parser'); const app = express(); const PORT = 3000; // You can change this port if needed // Middleware app.use(bodyParser.json()); // POST route to receive image data app.post('/', (req, res) => { const imageData = req.body.image; // Process the image data as needed console.log('Received image data:', imageData); res.send('Image received successfully!'); }); // Start the server app.listen(PORT, () => { console.log(`Server is running on port ${PORT}`); });
